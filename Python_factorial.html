Вычисление факториала числа 10 (10!):

def factorial(n):
            if n < 0:
                raise ArithmeticError('Факториал отрицательного числа.')
            f = 1
            for i in range (2, n + 1):
                f *= i
            return f
        
        print(factorial(10)) # 3628800
